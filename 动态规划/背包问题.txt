 int maxValue(vector<int> w, vector<int> v, int n, int cap) 
{
        // write code here
        int dp[1000][1000];
        int i,j;
        for(i=0;i<=n;i++)
            dp[i][0]=0;
        for(j=0;j<=cap;j++)
            dp[0][j]=0;
        for(i=1;i<=n;i++)
            for(j=1;j<=cap;j++)
        {
            if(j-w[i-1]<0)
                dp[i][j]=dp[i-1][j];
            else
                dp[i][j]=max(dp[i-1][j],dp[i-1][j-w[i-1]]+v[i-1]);
        }              
        return dp[n][cap]; 
    }

    int maxValue(vector<int> w, vector<int> v, int n, int cap) {
        int f[1003];
        for(int i = 1;i <= cap;++ i) f[i] = -1;
        f[0] = 0;
        for(int i = 0;i < n;++ i)
            for(int j = cap;j >= w[i];-- j)
            if(f[j - w[i]] != -1)
                f[j] = max(f[j],f[j - w[i]] + v[i]);
        int ret = 0;
        for(int i = 0;i <= cap;++ i)
        if(f[i] > ret)
            ret = f[i];
        return ret;
    }
