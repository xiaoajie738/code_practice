#include<iostream>
#include<vector>
using namespace std;
//n颗相同的糖果，分给m个人，每人至少一颗，问有多少种分法。
//	给定n和m，请返回方案数，保证n小于等于12，且m小于等于n。
int getWays(int n, int m) 
{
	// write code here
	if(n < m || n <= 0 || m <= 0)
		return 0;
	--n;
	--m;
	int res = 1;
	for(int i = n-m+1; i <= n; i++)
		res *= i;
	for(int i = 2; i <= m; i++)
		res /= i;
	return res;
}
int main()
{
   
   cout << getWays(10,3);
   system("pause");
   return 0;
}
