#include<iostream>

int main()
{
	 
	  int m;
	  while(cin >> m)
	  {
	     vector<vector<int> > data(m);
		 for(int i = 0; i < m; ++i)
		 {
			 vector<int> tmp;
			 char a;
             cin >> a;
			 if(a == 'I')
			 {
				 tmp.push_back(-1);
				 int num;
				 cin >> num;
				 tmp.push_back(num);
			 }
			 else if(a == 'O')
			 {
				 tmp.push_back(0);
				 int num;
				 cin >> num;
				 tmp.push_back(num);
			 }
			 else
			 {
				 tmp.push_back(-2);
			 }
			 data[i] = tmp;		
		 }
		 //≈–∂œ «∑Ò∫œ∑®
		 vector<int> vec;
		 int res = -1;
		 for(int i = 0; i < m; ++i)
		 {
			 if(data[i][0] == -1)
				 vec.push_back(data[i][1]);
			 else if(data[i][0] == -2)
				 vec.push_back(-2);
			 else
			 {
				vector<int>::iterator s = find( vec.begin(), vec.end(),data[i][1]);
				if(s != vec.end())
				{
				   vec.erase(s);
				}
				else
				{
				   s = find( vec.begin(), vec.end(),-2);
				   if(s != vec.end())
				   {
					   vec.erase(s);
				   }
				   else
				   {
				       res = i+1;
					   break;
				   }
				}
				
			 }
		 }
		 cout << res << endl;
	  }
	  system("pause");
	  return 0;
}
