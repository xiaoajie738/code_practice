#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;


int main()
{
    int n; 
	while(cin >> n)
	{
		long long res[31] = {0};
	   res[0] = 3;
	   res[1] = 9;
	   for(int i = 2; i < n; i++)
	       res[i] = 2*res[i-1] + res[i-2];
	  cout << res[n-1] << endl;
	} 
	system("pause");
    return 0;
}
