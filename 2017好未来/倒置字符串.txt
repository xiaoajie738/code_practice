#include<iostream>
#include<string>
#include<vector>
#include<sstream>
#include<algorithm>
using namespace std;

//输入例子:
//I like  beijing.
//
//输出例子:
//beijing. like I
string ReverseSentence0(string str) 
{
	string res,tmp;
	for(unsigned int i = 0; i < str.size(); ++i)
	{
		if(str[i] == ' ') 
		{
			res = " " + tmp + res;
			tmp = "";
		}
		else 
			tmp += str[i];
	}
	if(tmp.size())
		res = tmp + res;
	return res;
}

int main()
{
	string str;
	while(getline(cin,str))
	{
	  cout << ReverseSentence0(str);
	}
	return 0;
}
