bool isInclude(char s, string str2)
{
   for(unsigned int i = 0; i < str2.size(); ++i)
   {
      if(s == str2[i])
		  return true;
   }
   return false;
}
string ReverseSentence0(string str1,string str2) 
{
	string res;
	for(unsigned int i = 0; i < str1.size(); ++i)
	{
		if(str1[i] == ' ') 
		{
			res += ' ';
			
		}
		else 
		{
			if(!isInclude(str1[i],str2))
				res += str1[i];
		}
			
	}
	return res;
}
