对于一个数组，请设计一个高效算法计算需要排序的最短子数组的长度。
给定一个int数组A和数组的大小n，请返回一个二元组，代表所求序列的长度。(原序列位置从0开始标号,若原序列有序，返回0)。保证A中元素均为正整数
//最短有序子数组练习题
int shortestSubsequence(vector<int> A, int n) 
{
	// write code here
	int max = A[0];
	int min = A[n-1];
	int leftPostion = 0, rightPostion = 0;
	//找最右边的位置
	for(int i = 1; i < A.size(); i++)
	{
	   if(A[i] >= max)
		   max = A[i];
	   else
		   rightPostion = i;
	}
	//找最左边的位置
	for(int i = n-2; i >= 0; i--)
	{
		if(A[i] <= min)
			min = A[i];
		else
			leftPostion = i;
	}
	if(rightPostion == leftPostion)
		return 0;
	return rightPostion - leftPostion + 1;
	 
}
